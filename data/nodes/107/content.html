Typing ref() 

import { ref } from 'vue'
import type { Ref } from 'vue'

const year: Ref<string | number> = ref('2020')
year.value = 2020 // ok!

//Or, by passing a generic argument when calling `ref()`
const year = ref<string | number>('2020')
year.value = 2020 // ok!
