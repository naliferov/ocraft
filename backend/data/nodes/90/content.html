class QueueSimplified {
  constructor() {
    this.front = null;  // Ссылка на первый элемент
    this.rear = null;   // Ссылка на последний элемент
    this.length = 0;    // Размер очереди
  }

  // Создание нового узла
  createNode(data) {
    return { data, next: null };
  }

  // Добавление элемента в конец очереди - O(1)
  enqueue(element) {
    const newNode = this.createNode(element);
    
    if (this.isEmpty()) {
      // Если очередь пуста, новый узел становится и первым, и последним
      this.front = newNode;
      this.rear = newNode;
    } else {
      // Добавляем новый узел в конец и обновляем указатель rear
      this.rear.next = newNode;
      this.rear = newNode;
    }
    
    this.length++;
    return this.length;
  }

  // Удаление и возврат первого элемента очереди - O(1)
  dequeue() {
    if (this.isEmpty()) {
      return null;
    }
    
    const removedNode = this.front;
    
    // Если это был последний элемент
    if (this.front === this.rear) {
      this.rear = null;
    }
    
    this.front = this.front.next;
    this.length--;
    
    return removedNode.data;
  }

  // Возврат первого элемента без удаления - O(1)
  peek() {
    if (this.isEmpty()) {
      return null;
    }
    return this.front.data;
  }

  // Проверка, пуста ли очередь - O(1)
  isEmpty() {
    return this.front === null;
  }

  // Возврат размера очереди - O(1)
  size() {
    return this.length;
  }

  // Очистка очереди - O(1)
  clear() {
    this.front = null;
    this.rear = null;
    this.length = 0;
  }

  // Преобразование очереди в строку для вывода - O(n)
  print() {
    if (this.isEmpty()) {
      return "Очередь пуста";
    }
    
    let result = [];
    let current = this.front;
    
    while (current) {
      result.push(current.data);
      current = current.next;
    }
    
    return result.join(", ");
  }
}

function queueExample() {
  const queue = new QueueSimplified();
  
  queue.enqueue("Первый");
  queue.enqueue("Второй");
  queue.enqueue("Третий");
  
  console.log("Очередь после добавления 3 элементов:");
  console.log(queue.print());
  console.log("Размер очереди:", queue.size());
  
  console.log("Первый элемент:", queue.peek());
  
  console.log("Удаленный элемент:", queue.dequeue());
  console.log("Очередь после удаления элемента:");
  console.log(queue.print());
  
  console.log("Очередь пуста?", queue.isEmpty());
  
  queue.enqueue("Четвертый");
  queue.enqueue("Пятый");
  console.log("Очередь после добавления новых элементов:");
  console.log(queue.print());
  
  // Очистка очереди
  queue.clear();
  console.log("Очередь после очистки:");
  console.log(queue.print());
  console.log("Очередь пуста?", queue.isEmpty());
}

// Запуск примера
queueExample();
